```
pip install Flask
pip install Flask-Login
pip install Flask-SQLAlchemy
pip install Flask-Bcrypt
pip install Flask-WTF
pip install pywin32
pip install email-validator
pip install pyperclip
pip install pyautogui
pip install pygetwindow
```
